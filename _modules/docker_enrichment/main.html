[docs]async def get_ticker_names_and_id(sector, table):
    """
    Get ticker names based on sectors from a given table.

    Args:
        sectors (list): List of sectors to filter ticker names. If 'ALL' is included, ticker names for all sectors are returned.
        table: The table object to scan for ticker names.

    Returns:
        list: List of ticker names based on the specified sectors.
    """
    ticker_name = []
    item_list = [item async for item in table.read_all_items()]
    if sector == 'ALL':
        for doc in item_list:
            ticker_name.append(doc.get('Ticker_Name'))
    else:
        for doc in item_list:
            if doc.get('Sector') == sector:
                ticker_name.append(doc.get('Ticker_Name'))
    return ticker_name

[docs]def get_sentiment_class(news_sent_4, news_sent_5, news_sent_6):
    """
    Determine the sentiment class based on the given sentiment scores.

    Args:
        neg (float): Negative sentiment score.
        pos (float): Positive sentiment score.
        neut (float): Neutral sentiment score.

    Returns:
        str: Sentiment class ('positive', 'negative', or 'neutral').
    """
    if news_sent_4 > news_sent_5 and news_sent_4 > news_sent_6:
        sentiment_class = 'negative'
    elif news_sent_5 > news_sent_4 and news_sent_5 > news_sent_6:
        sentiment_class = 'positive'
    elif news_sent_6 > news_sent_4 and news_sent_6 > news_sent_5:
        sentiment_class = 'neutral'
    return sentiment_class

    """
  Perform sentiment analysis on the input sentence.

  Args:
        sentence (str): Input sentence to analyze.

  Returns:
        dict: Sentiment analysis results including text, sentiment scores, sentiment class, and identified entities.
    """
    table = np.empty([0, 7])
    senti_sent = sp(sentence)
    scores2 = None
    try:
        for span in senti_sent.sents:
            sent = False
            for ent in span.ents:
                if ent.label_ == 'ORG':
                    if not sent:
                        scores2 = analyze_sentiment(span.text)
                        table = np.append(table, np.array([[id, ent.text, ent.label_, span.text, scores2[0][1], scores2[0][0], scores2[0][2]]]), axis=0)
                        sent = True
                    else:
                        table = np.append(table, np.array([[id, ent.text, ent.label_, span.text, scores2[0][1], scores2[0][0], scores2[0][2]]]), axis=0)
    except:
        pass
    entities = []
    unique_entity = []
    for news_sent in table:
        if news_sent[1] in unique_entity:
            continue
        else:
            unique_entity.append(news_sent[1])
            list_of_entity = {'entity': news_sent[1], 'entity_type': news_sent[2]}
            entities.append(list_of_entity)
        news_sentence = news_sent[3]
        negative_sentiment_score = news_sent[4]
        positive_sentiment_score = news_sent[5]
        neutral_sentiment_score = news_sent[6]
    try:
        sentiment_class = get_sentiment_class(news_sent[4], news_sent[5], news_sent[6])
        sentiment_analysis = {'text': news_sentence, 'negative_sentiment_score': negative_sentiment_score, 'positive_sentiment_score': positive_sentiment_score, 'neutral_sentiment_score': neutral_sentiment_score, 'sentiment_class': sentiment_class, 'entities': entities}
        return sentiment_analysis
    except:
        print('No ORG present in sentence')





[docs]def compute_summary(text):
    """
    Compute a summary of the input text using a pre-trained summarization model.

    Args:
        text (str): The input text to be summarized.

    Returns:
        str: The generated summary of the input text.
    """
    summary = ''
    try:
        preprocess_text = text.strip().replace('\\n', '')
        db_prepared_Text = 'summarize: ' + preprocess_text
        tokenized_text = summarization_tokenizer.encode(db_prepared_Text, return_tensors='pt', truncation=True)
        summary_db = summarization_model.generate(tokenized_text, num_beams=4, no_repeat_ngram_size=2, min_length=30, max_length=100, early_stopping=True)
        summary = summarization_tokenizer.decode(summary_db[0], skip_special_tokens=True)
        print('Summary :', summary)
    except Exception as e:
        print('Error in generating summary: ', e)
    return summary

    """
    Find sentences containing 'events' from the provided text.

    Args:
        text (str): Text from which we need to find desired 'events'.

    Returns:
        events (list of strings): Sentences containing 'events' as a list.
    """
    events = []
    text_sent = event_tokenizer(str(text))
    sentences_fb = []
    np_zeros = np.zeros((768,))
    for i in range(len(text_sent)):
        try:
            emb = finbert.sentence_vector(text_sent[i])
            np_arr = emb.cpu().detach().numpy()
            sentences_fb.append(np_arr)
        except Exception as e:
            print('Except: ', e)
            print('error while getting sentence embeddings')
            sentences_fb.append(np_zeros)
    if len(sentences_fb) == 0:
        print('except')
        sentences_fb.append(np_zeros)
    prediction_labels = event_model.predict(sentences_fb)
    for i in range(len(prediction_labels)):
        if prediction_labels[i] == 0:
            events.append(text_sent[i])
    return events

    """
    Take an input text sentence of 'events' and classify it into different pre-defined categories.

    The categories considered are: 'Debt', 'Dividend', 'Employment', 'Financial Results', 'Investment & Funding',
    'Litigation', 'Macroeconomics', 'Merger & Acquisition', 'Partnership & JointVenture', 'Products & Services',
    'Rating & Recommendation', 'Securities Turnover', and 'Share Repurchase'.

    Args:
        text (str): The input sentence as a string (e.g., a paragraph of relevant event sentences).

    Returns:
        lists: top_2_events and all_events
    """
    top_2_events = []
    all_events = []
    label_d = {0: 'Merger_and_Acquisition', 1: 'Partnership_and_JointVenture', 2: 'Litigation', 3: 'Investment_and_Funding', 4: 'ShareRepurchase', 5: 'Financial_Results', 6: 'Dividend', 7: 'Rating_and_Recommendation', 8: 'Securities_Turnover', 9: 'Debt', 10: 'Other'}
    events = find_events(text)
    for event in events:
        text2 = str(event)
        print('getting sentence embeddings using tfidf vectoriser...')
        text_emb = vect.transform([text2.encode('ascii', 'ignore').decode('ascii').lower()])
        print('starting multi label classification...')
        pred = clf.predict(text_emb)
        prob = clf._predict_proba_lr(text_emb)
        labels = [label_d[np.argmax(prob)]][0]
        top_labels = [label_d[i] for i in (-prob).argsort()[0]]
        all_probabilities = list(np.sort(prob[0])[::-1])
        dict_events = {'text': event, 'labels': labels, 'probability': np.max(prob)}
        top_2_events_dict = {'text': event, 'labels': top_labels[:2], 'probability': all_probabilities[:2]}
        top_2_events.append(top_2_events_dict)
        all_events_dict = {'text': event, 'labels': top_labels, 'probability': all_probabilities}
        all_events.append(all_events_dict)
    return (all_events, top_2_events)





[docs]def perform_enrichment(article):
    """
    Perform enrichment on the given article.
    Summarization, and Event Detection on the article.

    Args:
        article (str): The text of the article to perform enrichment on.

    Returns:
        dict: A dictionary containing the results of the enrichment process, including:
        "sentiment_analysis", "ner_analysis", "summary", "events"
    """
    batch = nltk.tokenize.sent_tokenize(article)
    sentiment_analysis_list = []
    summarization_text = []
    for item in batch:
        sentence_sentiment = get_sentiment(item)
        sentiment_analysis_list.append(sentence_sentiment)
    summarization = compute_summary(article)
    start_time = time.time()
    events_detected = find_mult_class(article)
    print('--- %s seconds ---' % (time.time() - start_time))
    events = {'all_events': events_detected[0], 'top_2_events': events_detected[1]}
    unique_ent = []
    dict_of_unique_ent = []
    for analysis in sentiment_analysis_list:
        if analysis == None:
            continue
        else:
            for object in analysis['entities']:
                if object['entity'] in unique_ent:
                    continue
                else:
                    unique_ent.append(object['entity'])
                    entities = {'entity': object['entity'], 'entity_type': object['entity_type']}
                    dict_of_unique_ent.append(entities)
    output = {'sentiment_analysis': str(sentiment_analysis_list), 'ner_analysis': str(dict_of_unique_ent), 'summary': str(summarization), 'events': str(events)}
    return output
