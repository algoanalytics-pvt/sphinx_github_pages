def write_json_file(file_name, data):
    """
    Write json object to json file
    Args:file_name (sting): filename (currenty ticker symbols)
    data (dict): dict as json object to write to file
    """
    session = boto3.session.Session()
    client = session.client(service_name='s3', region_name=REGION_NAME, aws_access_key_id=ALGO_ACCESS_TOKEN, aws_secret_access_key=ALGO_SECRET_ACCESS_TOKEN)
    client.put_object(Body=json.dumps(data), Bucket='algoanalytics-fabric-website', Key=file_name)
    print(f"Uploaded {file_name.split('/')[-1]} to S3")
