def get_news_distrib_across_days(json_data, ticker_name, output_dict):
    total_news_articles = len(json_data['news_dict'][ticker_name])
    news_dates = [json_data['news_dict'][ticker_name][x]['news_date'].split(' ')[0] for x in range(total_news_articles)]
    sorted_dates = sorted(news_dates, key=lambda x: datetime.strptime(x, '%Y-%m-%d'))
    counts = Counter(sorted_dates)
    x = list(counts.keys())
    y = list(counts.values())
    output_dict['news_distrib_across_days'] = {'x': x, 'y': y}
    return output_dict

def get_news_source_distrib(json_data, ticker_name, output_dict, limit=10):
    total_news_articles = len(json_data['news_dict'][ticker_name])
    news_sources = [json_data['news_dict'][ticker_name][x]['news_source'] for x in range(total_news_articles)]
    news_sources = pd.DataFrame(news_sources, columns=['news_sources'])
    news_sources = news_sources['news_sources'].value_counts().reset_index()
    cat = 'Others'
    value = news_sources.iloc[limit:]['count'].sum()
    news_sources = news_sources.iloc[:limit, :]
    if value > 0:
        news_sources = pd.concat([news_sources, pd.DataFrame({'news_sources': cat, 'count': value}, index=[10])])
    news_sources['%_count'] = news_sources['count'] / news_sources['count'].sum() * 100
    x = news_sources['news_sources'].tolist()
    y = news_sources['count'].tolist()
    output_dict['news_source_distrib'] = {'x': x, 'y': y}
    return output_dict

def get_sentiment_score(article):
    senti_score = 0
    for event in article['sentiment_analysis']:
        if event is None:
            continue
        if event['sentiment_class'] == 'negative':
            senti_score += -1
        elif event['sentiment_class'] == 'positive':
            senti_score += 1
        else:
            senti_score += 0
    return senti_score

def get_normalized_sentiment_score(article):
    pos_count = 0
    neg_count = 0
    for event in article['sentiment_analysis']:
        if event is None:
            continue
        if event['sentiment_class'] == 'negative':
            neg_count += 1
        elif event['sentiment_class'] == 'positive':
            pos_count += 1
        else:
            pass
    if pos_count == 0 and neg_count == 0:
        return 0
    else:
        return (pos_count - neg_count) / (pos_count + neg_count)

def get_avg_sent_score_across_days(json_data, ticker_name, output_dict):
    scores = {}
    total_news_articles = len(json_data['news_dict'][ticker_name])
    for i in range(total_news_articles):
        news_article = json_data['news_dict'][ticker_name][i]
        date = news_article['news_date'].split(' ')[0]
        if date in scores.keys():
            scores[date].append(get_normalized_sentiment_score(news_article))
        else:
            scores[date] = [get_normalized_sentiment_score(news_article)]
    avg_sent_score = {}
    for (key, values) in scores.items():
        average = sum(values) / len(values)
        avg_sent_score[key] = round(average, 2)
    avg_sent_score = dict(sorted(avg_sent_score.items(), key=lambda x: datetime.strptime(x[0], '%Y-%m-%d')))
    (x, y) = zip(*avg_sent_score.items())
    output_dict['avg_sent_across_days'] = {'x': x, 'y': y}
    return output_dict

def get_sentiment_stacked_barchart(json_data, ticker_name, output_dict):
    (dates, positives, negatives, neutral) = ([], [], [], [])
    total_news_articles = len(json_data['news_dict'][ticker_name])
    for i in range(total_news_articles):
        news_article = json_data['news_dict'][ticker_name][i]
        date = news_article['news_date'].split(' ')[0]
        senti_list = get_sentiment_lables(news_article)
        dates.append(date)
        positives.append(senti_list.count('positive'))
        negatives.append(senti_list.count('negative'))
        neutral.append(senti_list.count('neutral'))
    df = pd.DataFrame({'date': dates, 'positives': positives, 'negatives': negatives, 'neutral': neutral})
    df = df.groupby('date').sum().reset_index()
    dates = df['date'].tolist()
    pos_count = df['positives'].tolist()
    neg_count = df['negatives'].tolist()
    neutral_count = df['neutral'].tolist()
    output_dict['sentiment_stacked_barchart'] = {'dates': dates, 'pos_count': pos_count, 'neg_count': neg_count, 'neutral_count': neutral_count}
    return output_dict

def get_event_label_heatmap(json_data, ticker_name, output_dict):
    event_df = pd.DataFrame(columns=['Share Repurchase', 'Other', 'Partnership & JointVenture', 'Macroeconomics', 'Debt', 'Merger & Acquisition', 'Litigation', 'Securities Turnover', 'Employment', 'Rating & Recommendation', 'Dividend', 'Investment & Funding', 'Products & Services', 'Financial Results'])
    total_news_articles = len(json_data['news_dict'][ticker_name])

    def get_top_event(event):
        return event['labels'][0]
    for i in range(total_news_articles):
        news_article = json_data['news_dict'][ticker_name][i]
        date = news_article['news_date'].split(' ')[0]
        events = news_article['All_Events']
        if events == 'Null':
            continue
        for event in events:
            top_event = get_top_event(event)
            if date in event_df.index:
                event_df.loc[date, top_event] += 1
            else:
                event_df.loc[date, top_event] = 1
                event_df.fillna(0, inplace=True)
    output_dict['event_label_heatmap'] = event_df.sort_index().to_dict()
    return output_dict

def get_entity_tree_diagram(json_data, ticker_name, output_dict):
    top_entities = 50
    entity_dict = {}
    total_news_articles = len(json_data['news_dict'][ticker_name])
    for i in range(total_news_articles):
        news_article = json_data['news_dict'][ticker_name][i]
        for entity in news_article['ner_analysis']:
            if entity['entity'] in entity_dict.keys():
                entity_dict[entity['entity']] += 1
            else:
                entity_dict[entity['entity']] = 1
    sorted_entities = sorted(entity_dict.items(), key=lambda x: x[1], reverse=True)
    events = [event for (event, _) in sorted_entities][:top_entities]
    frequencies = [freq for (_, freq) in sorted_entities][:top_entities]
    total = sum(frequencies)
    sizes = [freq / total * 100 for freq in frequencies]
    output_dict['entity_tree_diagram'] = {'events': events, 'sizes': sizes}
    return output_dict

def get_ner_org_freq(news_output, ticker_name, output_dict):
    ner_output = []
    for i in range(len(news_output['news_dict'][ticker_name])):
        ner_record = news_output['news_dict'][ticker_name][i]['ner_org_freq']
        if len(ner_record) == 0:
            pass
        else:
            ner_output.append(ner_record)
    combined_dict = {}
    for sublist in ner_output:
        for item in sublist:
            entity = item.get('entity', '')
            frequency = item.get('frequency', 0)
            if entity in combined_dict:
                combined_dict[entity] += frequency
            else:
                combined_dict[entity] = frequency
    sorted_dict = dict(sorted(combined_dict.items(), key=lambda item: item[1], reverse=True))
    top_15_pairs = list(sorted_dict.items())[:15]
    output_dict['ner_org_freq'] = top_15_pairs
    return output_dict
