def Find_url(string):
    """ Finds if a url is present in a news article. This function is used for cleaning news articles 
    from Business Standard source, as Business standard returns unnecessary content while scraping 
    before the main article.

    Args: 
        string (str) : The scraped article from "business standard" news source.

    Returns: 
        element (str): First URL from the list of urls extracted from the scraped news article.
        
    """
    regex = '(?i)\\b((?:https?://|www\\d{0,3}[.]|[a-z0-9.\\-]+[.][a-z]{2,4}/)(?:[^\\s()<>]+|\\(([^\\s()<>]+|(\\([^\\s()<>]+\\)))*\\))+(?:\\(([^\\s()<>]+|(\\([^\\s()<>]+\\)))*\\)|[^\\s`!()\\[\\]{};:\'\\".,<>?«»“”‘’]))'
    url = re.findall(regex, string)
    return [x[0] for x in url]

def get_content(link):
    """ Uses "Article" function from "newspaper3k" library to scrape the news article.

    Args:
        link (str) : The link or url of the article to be scraped.

    Returns:
        text (str) : content of the link or url provided.
        
    """
    try:
        article = Article(link)
        article.download()
        article.parse()
    except:
        return False
    return article.text

def news_lang_check(description):
    """ Checks if the news article scraped is of english language.

    Args: 
        description (str) : scraped content of the news article

    Returns:
        bool: True if the language of the article is English, else False.

    CreatedBy: 
            Mansi Maurya <mmaurya@algoanalytics.com>
            
    LastUpdated: 
            07th June 2023

    ChangesMadeInCode:
            Created the function
            
    """
    printable_str = ''.join((x for x in description if x.isprintable()))
    (isReliable, textBytesFound, details, vector) = cld2.detect(printable_str, returnVectors=True)
    languages = ['TELUGU', 'HINDI', 'KOREAN', 'JAPANESE', 'CHINESE', 'SPANISH', 'GERMAN']
    if any((element[2] in languages for element in vector)):
        return False
    else:
        return True

def news_longDesc_check(description, key_words):
    """ Checks if certain keywords are present in the scraped news article.

    Args:
        description (str) : scraped content of the news article.
        keywords (str): string of keywords to check their presence in the scraped news article.

    Returns:
        bool: True if any one of the keywords are present in the scraped news article.

    CreatedBy: 
            Mansi Maurya <mmaurya@algoanalytics.com>
            
    LastUpdated: 
            04th July 2023

    ChangesMadeInCode:
            Created the function
            
    """
    keyword_list = key_words.split(',')
    for k in keyword_list:
        if k.lower() in description.lower():
            return True

async def add_news_article(news_table, news_id, ticker_symbol, item, long_description, keywords, count):
    """ Checks if the scraped news articles pass certain conditions like language check and keyword check 
    and on the basis of those results adds news articles to news table.

    Args: 
        news_table (AWS DynamoDB Table) : "News" Table from DynamoDB
        news_id (str) : Unique id which is assigned to every scraped article
        ticker_symbol (str) : ticker name of the company
        item (response of pygooglenews api) : The object returned of the pygooglenews api
        long_description (str): content of the article
        keywords (str) : keywords needed to check their presence in the content of the article
        count (int) : the number of articles inserted in the table

    Returns: 
        count (int) : the updated number of articles inserted in the table

    LastUpdated: 
            04th July 2023

    ChangesMadeInCode:
            Added keyword_passed condition in if statement, elif conditions code block for ITC Ltd, State Bank of India, Mahindra & Mahindra Limited.
            
    """
    try:
        long_desc = news_lang_check(long_description)
        keyword_passed = news_longDesc_check(long_description, keywords)
        if long_desc == True and keyword_passed == True:
            news = {'news_month': news_month, 'news_ID': news_id, 'crawled_date': (datetime_now + datetime.timedelta(hours=5, minutes=30)).strftime('%Y-%m-%d %H:%M:%S'), 'crawled_source': 'pygooglenews', 'news_for': ticker_symbol, 'news_source': item.source['title'], 'news_date': str((datetime.datetime.strptime(item.published, '%a, %d %b %Y %H:%M:%S GMT') + datetime.timedelta(hours=5, minutes=30)).strftime('%Y-%m-%d %H:%M:%S')), 'link': item.link, 'title': item.title, 'long_description': long_description, 'short_description': w3lib.html.remove_tags(item.summary), 'news_ner_flag': False, 'events_detection_flag': False}
            if 'Container Store' in news['title']:
                pass
                return count
            elif news['news_for'] == 'ITC Ltd':
                l = 'International Trade,input tax credit,ITC Sports Travel,ITC Midwest,International Tower Crane,Wyoming,Intercontinental Terminals Co,ITC tower,investment tax credits,ITC Latam,ITC Crypto'
                res = news_longDesc_check(long_description, l)
                if res == True:
                    pass
                    return count
                else:
                    news_table.put_item(Item=news)
                    count = count + 1
                    return count
            elif news['news_for'] == 'State Bank of India':
                l = 'Society of Breast Imaging,SBI Holdings,Shinsei Bank,State Bureau of Investigation,SBI VC Trade,SBI Group,Sourcebio International'
                if item.source['title'] == 'SBI Soccer':
                    pass
                    return count
                else:
                    res = news_longDesc_check(long_description, l)
                    if res == True:
                        pass
                        return count
                    else:
                        news_table.put_item(Item=news)
                        count = count + 1
                        return count
            elif news['news_for'] == 'Mahindra & Mahindra Limited':
                l = "M&M Bank,M&M Grocery,M&M's,M&M store,M&Ms,M&M International,M&M Area Community Foundation,M&M BBQ Company,M&M Carnot,M&M Food Mart,Skittles"
                res = news_longDesc_check(long_description, l)
                if res == True:
                    pass
                    return count
                else:
                    news_table.put_item(Item=news)
                    count = count + 1
                    return count
            else:
                news_table.put_item(Item=news)
                count = count + 1
                return count
        else:
            pass
            return count
    except Exception as e:
        print('Exception in add_news_article function: ', e)

def crawl_news_by_keyword(news_table, frequency, ticker_news_keywords, ticker_symbol, longDesc_keywords, text_search):
    """ Main Function for crawling news using pygooglenews library

    Args:
        news_table (AWS DynamoDB Table) : "News" Table from DynamoDB
        frequency (str) : The time duration from which the user wants to get the news
        ticker_news_keywords (str) : The query string which has to be passed to google news which will return news about the specific company.
        ticker_symbol (str) :  The string which will be used to store news along with the Ticker name in DynamoDB Table.
        longDesc_keywords (str) : Keywords that will be used to search them in the scraped news article
        text_search (str) : Value that is used to decide if to find news using text if the results returned by finding news using title is 0

    LastUpdated: 
            05th July 2023

    ChangesMadeInCode:
            Added if code blocks that runs when the article_text is not equal to None when using beautifulsoup and requests.
            
    """
    gn = GoogleNews(lang='en')
    result = gn.search(f'allintitle:{ticker_news_keywords}', when=frequency)
    if len(result['entries']) == 0 and text_search == '1':
        print(ticker_symbol)
        print('Seaching in the text')
        result = gn.search(f'{ticker_news_keywords}', when=frequency)
    newsitem = result['entries']
    c = 0
    for item in newsitem:
        news_id = ticker_symbol + ' +UNIQUE_T_AND_N_ID_SEPARATOR+ ' + item.link
        try:
            long_description = get_content(item.link)
            if long_description == '' or long_description == False:
                if item.source['title'] == 'Indiainfoline':
                    news = requests.get(item.link)
                    soup = BeautifulSoup(news.content, 'html.parser')
                    article_text = soup.find(class_='widget-content cms-content ng-star-inserted')
                    if article_text != None:
                        long_description = article_text.text
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'CNBCTV18':
                    news_class_pattern = re.compile('.*(art-page-wrapper|articleWrap).*')
                    news = requests.get(item.link)
                    soup = BeautifulSoup(news.content, 'html.parser')
                    article_text = soup.find('div', {'class': news_class_pattern})
                    if article_text:
                        for section in article_text.select('section'):
                            section.extract()
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'Press Trust of India':
                    news = requests.get(item.link)
                    soup = BeautifulSoup(news.content, 'html.parser')
                    article_text = soup.find(class_='big-image-description')
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'The Tribune India':
                    req = Request(item.link, headers={'User-Agent': 'Mozilla/5.0'})
                    webpage = urlopen(req).read()
                    soup = BeautifulSoup(webpage, 'html.parser')
                    article_text = soup.find(class_='story-desc')
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'Business Standard':
                    news = requests.get(item.link)
                    soup = BeautifulSoup(news.content, 'html.parser')
                    article_text = soup.find(class_='storycontent')
                    if article_text:
                        for tag in article_text.select('div.storycontent > div:nth-child(1) > div.mb-20'):
                            tag.extract()
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'Outlook India':
                    req = Request(item.link, headers={'User-Agent': 'Mozilla/5.0'})
                    webpage = urlopen(req).read()
                    soup = BeautifulSoup(webpage, 'html.parser')
                    article_text = soup.find(id='articleBody')
                    if article_text:
                        for tag in article_text.select('#articleBody > div.__related_stories_thumbs'):
                            tag.extract()
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'Investing.com India':
                    req = Request(item.link, headers={'User-Agent': 'Mozilla/5.0'})
                    webpage = urlopen(req).read()
                    soup = BeautifulSoup(webpage, 'html.parser')
                    article_text = soup.find(id='article-item-content')
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
                elif item.source['title'] == 'Economic Times':
                    req = Request(item.link, headers={'User-Agent': 'Mozilla/5.0'})
                    webpage = urlopen(req).read()
                    soup = BeautifulSoup(webpage, 'html.parser')
                    article_text = soup.find(class_='artData clr')
                    if article_text:
                        for tag in article_text.select('div.mb20.flt'):
                            tag.extract()
                    if article_text != None:
                        long_description = article_text.get_text().strip()
                        loop = asyncio.get_event_loop()
                        c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
            else:
                loop = asyncio.get_event_loop()
                c = loop.run_until_complete(add_news_article(news_table, news_id, ticker_symbol, item, long_description, longDesc_keywords, c))
        except Exception as e:
            print('Exception in crawl_news_by_keyword function: ', e)
    print(f'Number of news for {ticker_symbol} entered in the table : ', c)
