def identify_single_candlestick_patterns(row):
    print('Function called!')
    if row['Detected Patterns'] == 'Indecisive':
        if row['Open'] == row['Close'] and row['High'] > row['Open'] and (row['Low'] < row['Open']):
            pattern = 'Doji'
        elif row['Open'] == row['Close'] and row['High'] > row['Open'] and (row['Low'] > row['Open']):
            pattern = 'Gravestone Doji'
        elif row['Open'] == row['Close'] and row['High'] < row['Open'] and (row['Low'] < row['Open']):
            pattern = 'Dragonfly Doji'
        elif row['Open'] > row['Close'] and row['Open'] - row['Close'] > 2 * (row['High'] - row['Open']):
            pattern = 'Bearish Hammer'
        elif row['Open'] < row['Close'] and row['Close'] - row['Open'] > 2 * (row['Open'] - row['Low']):
            pattern = 'Bullish Hammer'
        elif row['Open'] > row['Close'] and row['Open'] - row['Close'] < 0.5 * (row['High'] - row['Low']):
            pattern = 'Bearish Spinning Top'
        elif row['Open'] < row['Close'] and row['Close'] - row['Open'] < 0.5 * (row['High'] - row['Low']):
            pattern = 'Bullish Spinning Top'
        elif row['Open'] == row['Close'] and row['High'] > row['Open'] and (row['Low'] == row['Open']):
            pattern = 'Hanging Man'
        elif row['Open'] == row['Close'] and row['High'] == row['Open'] and (row['Low'] < row['Open']):
            pattern = 'Shooting Star'
        elif row['Open'] < row['Close'] and row['Open'] == row['Low'] and (row['Close'] == row['High']):
            pattern = 'RainDrop'
        return pattern
    return row['Detected Patterns']

def handle_indecisive_patterns(row):
    if row['Detected Patterns'] == 'Indecisive':
        pattern = row['Detected Patterns']
        if pattern != 'Indecisive':
            standardized_pattern = pattern.lower().replace(' ', '')
            description = pattern_description.get(standardized_pattern, 'Description not available')
            return description
    return row['Pattern Descriptions']
